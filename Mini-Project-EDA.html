# GPA, GRE, TOEFL score for admitted students and non-admitted applicants
fig = plt.figure()
fig.set_figheight(10) # optional setting the height of the image
fig.set_figwidth(20)
plt.subplot(1, 3, 1)
sns.boxplot(y="GPA", x="OfferOfAdmissionExtended", data=df, palette="Set3")
sns.swarmplot(y="GPA", x="OfferOfAdmissionExtended", data=df, color=".25")
plt.title('Offer of admission and GPA')
plt.subplot(1, 3, 2)
sns.boxplot(y="GRE", x="OfferOfAdmissionExtended", data=df[df['GRE'] != 50], palette="Set1")
sns.swarmplot(y="GRE", x="OfferOfAdmissionExtended", data=df[df['GRE'] != 50], color=".25")
plt.title('Offer of admission and GRE')
plt.subplot(1, 3, 3)
sns.boxplot(y="TOEFL", x="OfferOfAdmissionExtended", data=df,  palette="Set2")
sns.swarmplot(y="TOEFL", x="OfferOfAdmissionExtended", data=df, color=".25")
plt.title('Offer of admission and TOEFL')
plt.show()
